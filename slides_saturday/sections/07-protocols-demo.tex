\begin{frame}{Resuscitation protocol (\sats{triageRed}{Red})}
  \textbf{Target response time: 0 minutes (immediate)}

  \begin{enumerate}
    \item \textbf{Secure airway \& vitals}
    \begin{itemize}
      \item Airway patency; continuous cardiac, SpO$_2$, NIBP monitoring
    \end{itemize}
    \item \textbf{Establish access}
    \begin{itemize}
      \item Two large-bore IVs or IO access
      \item Stat labs: VBG, lactate, cardiac enzymes
    \end{itemize}
    \item \textbf{Targeted intervention}
    \begin{itemize}
      \item ACLS / ATLS protocols as indicated
      \item Defibrillation, epinephrine, massive transfusion, etc.
    \end{itemize}
  \end{enumerate}

  \plain{Red bypasses registration and payment --- Code Red activated immediately upon fusion assignment.}
\end{frame}
